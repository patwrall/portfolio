A CUDA kernel, to check `reinterpret_cast`, pointer types, and CUDA-specific qualifiers/built-ins:

```cpp
// single-line comment
__global__ void sgemm_vectorized(int M, int N, int K, float alpha,
                                  const float* A, const float* B,
                                  float beta, float* C) {
    __shared__ float As[128];
    __shared__ float Bs[128];

    const uint threadRow = threadIdx.x / 8;
    const uint threadCol = threadIdx.x % 8;

    float4 tmp = reinterpret_cast<float4*>(&A[threadRow * K])[0];
    As[threadCol] = tmp.x;

    float threadResults[8] = {0.0f};

    for (int dotIdx = 0; dotIdx < 8; ++dotIdx) {
        float regA = As[dotIdx];
        for (int i = 0; i < 8; ++i) {
            threadResults[i] += regA * Bs[dotIdx * 8 + i];
        }
    }

    reinterpret_cast<float4*>(&C[threadRow * N + threadCol * 4])[0] =
        reinterpret_cast<float4*>(threadResults)[0];
}
```
